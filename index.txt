// 1. Index หลักสำหรับ Dashboard (Q1, Q3, Q4)
// ใช้สำหรับกวาดข้อมูลทั้งประเทศหรือจังหวัด แยกตามประเภทบัตรและสถานะ
db.vote_transactions.createIndex(
  { ballot_type: 1, integrity_status: 1, station_id: 1, sequence_no: -1 },
  { name: "idx_dashboard_agg" }
);

// 2. Index สำหรับเจาะจงหน่วยเลือกตั้ง (Q2, Q5, Q6)
// ใช้สำหรับการ Lookup, การหา submission ล่าสุดของหน่วย, และ Audit Trail
db.vote_transactions.createIndex(
  { station_id: 1, ballot_type: 1, sequence_no: -1 },
  { name: "idx_station_lookup" }
);

// 3. Index เสริมสำหรับ Audit Trail แบบรวม (ตอบโจทย์ Q5 แบบดูรวมทุกประเภทบัตร)
db.vote_transactions.createIndex(
  { station_id: 1, sequence_no: 1 },
  { name: "idx_audit_trail" }
);
